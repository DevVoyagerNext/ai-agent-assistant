# 大模型上下文记忆管理优化方案

## 1. 现状分析

### 1.1 核心数据结构

**会话表 (model.Session)**
```go
type Session struct {
    ID                   int64
    UserID               int64
    Title                string
    Summary              string          // 会话摘要
    LastSummaryMessageID int64          // 最后一次摘要的消息ID
    ModelID              string
    IsDeleted            bool
    UpdatedAt            time.Time
    CreatedAt            time.Time
}
```

**消息表 (model.Message)**
```go
type Message struct {
    ID         int64
    SessionID  int64
    ParentID   *int64
    Role       string    // user/assistant
    Content    string
    Status     string    // active
    TokensUsed int       // 未实际使用
    CreatedAt  time.Time
}
```

### 1.2 现有记忆管理机制

#### 短期记忆（chat_context）
- 获取最近 **4轮对话**（8条消息）
- 通过 `buildConversationMessages` 转换为AI格式
- 存储位置：内存中的 `historyMsgs` 变量

#### 长期记忆（session summary）
- `Session.Summary`: 会话摘要
- `Session.LastSummaryMessageID`: 记录摘要覆盖的消息范围
- 触发条件：用户消息达到4条后自动生成

#### 记忆更新逻辑（finalizeChatSideEffects）
1. 新会话：异步生成标题
2. 老会话：
   - 统计自上次摘要后的用户消息数
   - 达到4条时触发摘要更新
   - 摘要字数控制在500字以内

## 2. 存在的问题

### 2.1 核心问题
1. **固定窗口大小**：硬编码8条消息，无法根据Token预算动态调整
2. **Token管理缺失**：未统计消息Token数，可能超出模型上下文限制
3. **摘要质量不可控**：500字限制由模型承诺，缺乏强制验证
4. **上下文无相关性**：纯时间倒序，未考虑话题重要性、衰减

### 2.2 性能问题
5. **数据库查询优化**：`GetSessionMessages` 和 `finalizeChatSideEffects` 缺少必要索引
6. **并发竞态**：摘要更新为异步，可能多goroutine并发修改同一摘要

### 2.3 功能缺失
7. **无分层摘要**：长对话无法实现摘要的摘要
8. **无跨会话记忆**：无法利用用户偏好、历史偏好
9. **无监控指标**：缺少Token使用、摘要质量等监控

### 2.4 配置固化
10. **参数硬编码**：历史消息数、摘要触发阈值等均写死在代码中

## 3. 优化方案设计

### 3.1 整体架构

```

│                     Chat Request                         │
└─────────────────────┬───────────────────────────────────┘

```

### 3.2 核心改进点

1. **动态Token预算管理**
   - 配置化Token预算上限
   - 实时计算消息Token数
   - 按预算动态调整历史消息数量

2. **相关性评分**
   - 基于TF-IDF/Embedding的消息相似度
   - 时间衰减因子
   - 优先保留高相关性消息

3. **分层摘要**
   - 基础摘要（500字）
   - 深度摘要（对话过长时）
   - 摘要压缩策略

4. **配置化策略**
   - 全局配置：Token预算、摘要阈值
   - 会话级配置：可覆盖全局策略

5. **质量保证**
   - 摘要长度强制验证
   - 摘要质量评分（语义完整性）
   - 失败重试机制

6. **监控与优化**
   - Token使用统计
   - 上下文命中率
   - 摘要质量指标

```go
// ContextManager 上下文管理器
type ContextManager struct {
    db      *gorm.DB
    config  *config.ContextConfig
    tokenizer TokenCounter
}

// TokenCounter Token计数器接口
type TokenCounter interface {
    Count(text string) (int, error)
}

// ContextResult 上下文获取结果
type ContextResult struct {
    SystemMessage  *schema.Message
    Messages       []*schema.Message
    Summary        string
    TotalTokens    int
    ExcludedHints  []string  // 排除原因提示
}

// ConfigurableStrategy 可配置策略
type ConfigurableStrategy struct {
    MaxHistoryRounds     int     // 最大历史轮数
    MinRelevanceScore    float64 // 最小相关性分数
    TokenBudget          int     // Token预算
    SummaryThreshold     int     // 摘要触发阈值（用户消息数）
    EnableRelevance      bool    // 是否启用相关性筛选
    TimeDecayWeight      float64 // 时间衰减权重
}

// NewContextManager 创建上下文管理器
func NewContextManager(db *gorm.DB, config *config.ContextConfig, tokenizer TokenCounter) *ContextManager {
    return &ContextManager{
        db:         db,
        config:     config,
        tokenizer:  tokenizer,
    }
}

// BuildChatContext 构建聊天上下文
func (cm *ContextManager) BuildChatContext(ctx context.Context, sessionID int64, userMessage string, latestMessageCount int, strategy *ConfigurableStrategy) (*ContextResult, error) {
    // 1. 获取会话信息
    var session model.Session
    if err := cm.db.WithContext(ctx).Where("id = ?", sessionID).First(&session).Error; err != nil {
        return nil, errors.New("会话不存在")
    }

    // 2. 获取候选消息
    candidateMsgs, err := cm.getCandidateMessages(ctx, sessionID, strategy.MaxHistoryRounds)
    if err != nil {
        return nil, err
    }

    // 3. 计算Token并筛选消息
    selectedMsgs, hints, err := cm.selectMessagesByBudget(candidateMsgs, userMessage, strategy)
    if err != nil {
        return nil, err
    }

    // 4. 构建系统消息（含摘要）
    systemMsg := cm.buildSystemMessage(session, strategy)

    // 5. 转换为Eino格式
    messages := cm.convertToEinoMessages(systemMsg, selectedMsgs)

    // 6. 统计Token使用
    totalTokens := cm.countConversationToken(systemMsg, messages, strategy)

    return &ContextResult{
        SystemMessage: systemMsg,
        Messages:      messages,
        Summary:       session.Summary,
        TotalTokens:   totalTokens,
        ExcludedHints: hints,
    }, nil
}

// getCandidateMessages 获取候选消息（基础的历史消息获取）
func (cm *ContextManager) getCandidateMessages(ctx context.Context, sessionID int64, maxRounds int) ([]model.Message, error) {
    var messages []model.Message
    limit := maxRounds * 2 // 每轮2条（user+assistant）

    if err := cm.db.WithContext(ctx).
        Where("session_id = ? AND status = 'active'", sessionID).
        Order("created_at desc").
        Limit(limit).
        Find(&messages).Error; err != nil {
        return nil, err
    }

    // 反转为正序
    for i := len(messages)/2 - 1; i >= 0; i-- {
        opp := len(messages) - 1 - i
        messages[i], messages[opp] = messages[opp], messages[i]
    }

    return messages, nil
}

// selectMessagesByBudget 根据Token预算和相关性选择消息
func (cm *ContextManager) selectMessagesByBudget(messages []model.Message, userMessage string, strategy *ConfigurableStrategy) ([]model.Message, []string, error) {
    var selected []model.Message
    var hints []string
    usedTokens := 0

    // 计算新消息的Token数
    newUserMsgToken, _ := cm.tokenizer.Count(userMessage)
    budget := strategy.TokenBudget - newUserMsgToken // 预留用户消息Token

    // 相关性评分
    scoredMsgs := cm.scoreMessages(messages, userMessage, strategy)

    // 按分数排序并选择
    for _, scoredMsg := range scoredMsgs {
        msgToken, _ := cm.tokenizer.Count(scoredMsg.Message.Content)
        if usedTokens+msgToken > budget {
            hints = append(hints, fmt.Sprintf("超出Token预算，已忽略更早的对话"))
            continue
        }
        if strategy.EnableRelevance && scoredMsg.Score < strategy.MinRelevanceScore {
            hints = append(hints, fmt.Sprintf("相关性较低，已忽略部分对话"))
            continue
        }
        selected = append(selected, scoredMsg.Message)
        usedTokens += msgToken
    }

    if len(selected) == 0 {
        hints = append(hints, "由于Token限制，未包含历史对话")
    }

    return selected, hints, nil
}

// MessageScored 带分数的消息
type MessageScored struct {
    Message  model.Message
    Score    float64
}

// scoreMessages 计算消息相关性分数
func (cm *ContextManager) scoreMessages(messages []model.Message, userMessage string, strategy *ConfigurableStrategy) []MessageScored {
    // 1. 基础分数：基于TF-IDF或Embedding的相似度
    baseScores := cm.calculateSimilarity(messages, userMessage)

    // 2. 时间衰减：越新的消息权重越高
    now := time.Now()
    scored := make([]MessageScored, len(messages))
    for i, msg := range messages {
        timeDiff := now.Sub(msg.CreatedAt).Hours()
        decay := 1.0 / (1.0 + strategy.TimeDecayWeight*timeDiff/24.0) // 以天为单位

        finalScore := baseScores[i] * decay
        scored[i] = MessageScored{
            Message: msg,
            Score:   finalScore,
        }
    }

    // 3. 按分数排序（保留消息对）
    return cm.sortPreservingPairs(scored)
}

// sortPreservingPairs 保留消息对的排序
func (cm *ContextManager) sortPreservingPairs(scored []MessageScored) []MessageScored {
    // 将消息配对（user+assistant），按assistant的分数排序
    // 实现省略...

}

// buildSystemMessage 构建系统消息
func (cm *ContextManager) buildSystemMessage(session model.Session, strategy *ConfigurableStrategy) *schema.Message {
    systemPrompt := cm.getBaseSystemPrompt()

    // 添加摘要
    if session.Summary != "" {
        systemPrompt += fmt.Sprintf("\n\n【历史对话摘要】\n%s", session.Summary)
    }

    // 添加深度摘要（如果存在）
    if session.DeepSummary != "" {
        systemPrompt += fmt.Sprintf("\n\n【长期背景】\n%s", session.DeepSummary)
    }

    return schema.SystemMessage(systemPrompt)
}

// convertToEinoMessages 转换为Eino消息格式
func (cm *ContextManager) convertToEinoMessages(systemMsg *schema.Message, messages []model.Message) []*schema.Message {
    result := []*schema.Message{systemMsg}

    for _, msg := range messages {
        result = append(result, &schema.Message{
            Role:    schema.RoleType(msg.Role),
            Content: msg.Content,
        })
    }

    return result
}

// countConversationToken 统计对话总Token数
func (cm *ContextManager) countConversationToken(systemMsg *schema.Message, messages []*schema.Message, strategy *ConfigurableStrategy) int {
    total := 0
    for _, msg := range messages {
        tokens, _ := cm.tokenizer.Count(msg.Content)
        total += tokens
    }
    return total
}
```

### Task 2: 实现Token计数器

**文件**: `pkg/context_manager/token_counter.go`

```go
package context_manager

import (
    "regexp"
    "strings"
)

var (
    // 中文字符正则
    chineseChar = regexp.MustCompile(`[\p{Han}]`)
    // 英文单词正则
    englishWord = regexp.MustCompile(`[a-zA-Z0-9]+`)
    // 代码标记
    codeBlock = regexp.MustCompile("```[\\s\\S]*?```")
)

// GPT4Tokenizer GPT-4风格的Token估算器
type GPT4Tokenizer struct {
}

// NewGPT4Tokenizer 创建GPT-4风格Token估算器
func NewGPT4Tokenizer() *GPT4Tokenizer {
    return &GPT4Tokenizer{}
}

// Count 估算文本Token数
func (t *GPT4Tokenizer) Count(text string) (int, error) {
    if text == "" {
        return 0, nil
    }

    // 1. 计算中文字符Token（每个中文字符约1Token）
    chineseChars := chineseChar.FindAllString(text, -1)
    chineseTokens := len(chineseChars)

    // 2. 计算英文单词Token（每个单词约0.75Token）
    englishWords := englishWord.FindAllString(text, -1)
    englishTokens := int(float64(len(englishWords)) * 0.75)

    // 3. 添加基础开销（标点、空格等）
    baseTokens := int(float64(len(text)) * 0.02)

    // 4. 代码块特殊处理（代码Token密度较低，按1.5倍计算）
    codeBlocks := codeBlock.FindAllString(text, -1)
    codeExtra := 0
    for _, block := range codeBlocks {
        codeExtra += len(block) / 4 // 代码块每4字符约1Token
    }

    totalTokens := chineseTokens + englishTokens + baseTokens + codeExtra

    // 确保至少返回1（非空文本）
    if totalTokens == 0 && len(text) > 0 {
        totalTokens = 1
    }

    return totalTokens, nil
}

// EstimateTokens 估算Token数（简化版）
func EstimateTokens(text string) int {
    // 粗略估算：4个字符约1Token
    return len(text) / 4
}
```

```go
package service

import (
    "backend/global"
    "backend/model"
    "backend/pkg/context_manager"
    "context"
    "encoding/json"
    "errors"
    "fmt"
    "strconv"

    "time"

    "github.com/cloudwego/eino/schema"
    "github.com/google/uuid"
    "go.uber.org/zap"
    "gorm.io/gorm"
)

// SummaryService 摘要服务
type SummaryService struct {
    aiService     *AIService
    tokenizer     context_manager.TokenCounter
    config        *SummaryConfig
}

// SummaryConfig 摘要配置
type SummaryConfig struct {
    MaxSummaryTokens   int           `json:"maxSummaryTokens"`  // 摘要最大Token数
    MinSummaryTokens   int           `json:"minSummaryTokens"`  // 摘要最小Token数
    TriggerThreshold   int           `json:"triggerThreshold"`  // 触发摘要的用户消息数
    EnableDeepSummary  bool          `json:"enableDeepSummary"` // 是否启用深度摘要
    DeepSummaryAfter   int           `json:"deepSummaryAfter"`  // 多少次摘要后生成深度摘要
    RetryTimes         int           `json:"retryTimes"`        // 重试次数
    RetryInterval      time.Duration `json:"retryInterval"`     // 重试间隔
}

// SummaryGenerationTask 摘要生成任务
type SummaryGenerationTask struct {
    SessionID        int64
    LastMessageID    int64
    PreviousSummary  string
    PreviousVersion  int
    TaskID           string
}

// NewSummaryService 创建摘要服务
func NewSummaryService(aiService *AIService, tokenizer context_manager.TokenCounter, config *SummaryConfig) *SummaryService {
    return &SummaryService{
        aiService: aiService,
        tokenizer: tokenizer,
        config:    config,
    }
}

// ShouldUpdateSummary 判断是否需要更新摘要
func (ss *SummaryService) ShouldUpdateSummary(ctx context.Context, db *gorm.DB, session model.Session, latestUserMsgID int64) bool {
    var count int64
    query := db.Model(&model.Message{}).
        Where("session_id = ? AND role = ?", session.ID, string(schema.User))

    if session.LastSummaryMessageID > 0 {
        query = query.Where("id > ?", session.LastSummaryMessageID)
    }

    // 使用新的索引增强查询性能
    query.Count(&count)

    return count >= int64(ss.config.TriggerThreshold)
}

// GenerateSummary 生成摘要
func (ss *SummaryService) GenerateSummary(ctx context.Context, task SummaryGenerationTask) (string, int, error) {
    var msgsToSummarize []model.Message
    q := global.GVA_DB.Where("session_id = ? AND status = 'active'", task.SessionID)
    if task.PreviousVersion > 0 {
        q = q.Where("id > ?", task.LastMessageID)
    }
    q.Order("created_at asc").Find(&msgsToSummarize)

    if len(msgsToSummarize) == 0 {
        return "", 0, errors.New("无新消息需要摘要")
    }

    // 构建摘要Prompt
    contentToSummarize := ss.buildSummaryPrompt(task.PreviousSummary, msgsToSummarize)

    // 生成摘要（带重试）
    summary, err := ss.generateWithRetry(ctx, contentToSummarize)
    if err != nil {
        return "", 0, fmt.Errorf("摘要生成失败: %w", err)
    }

    // 验证摘要质量
    if err := ss.validateSummary(summary); err != nil {
        global.GVA_LOG.Warn("摘要质量验证失败", zap.String("summary", summary), zap.Error(err))
        // 降级处理：缩短摘要
        summary = ss.fallbackSummary(summary)
    }

    // 计算Token数
    tokenCount, _ := ss.tokenizer.Count(summary)

    return summary, tokenCount, nil
}

// buildSummaryPrompt 构建摘要Prompt
func (ss *SummaryService) buildSummaryPrompt(previousSummary string, msgs []model.Message) string {
    var builder strings.Builder

    if previousSummary != "" {
        builder.WriteString("【之前的对话背景摘要】\n")
        builder.WriteString(previousSummary)
        builder.WriteString("\n\n")
    }

    builder.WriteString("【最新的对话记录】\n")
    for _, m := range msgs {
        roleName := "用户"
        if m.Role == string(schema.Assistant) {
            roleName = "AI"
        }
        builder.WriteString(fmt.Sprintf("[%s]: %s\n", roleName, m.Content))
    }

    // 添加Token限制
    maxChars := ss.config.MaxSummaryTokens * 4 // 粗略估算Token->字符
    builder.WriteString(fmt.Sprintf("\n请结合前文摘要和最新对话，重新生成一份全局简明摘要。")
    builder.WriteString(fmt.Sprintf("提取核心背景、用户意图和关键结论，字数严格控制在%d字以内！", maxChars))

    return builder.String()
}

// generateWithRetry 带重试的摘要生成
func (ss *SummaryService) generateWithRetry(ctx context.Context, prompt string) (string, error) {
    var lastErr error
    for i := 0; i < ss.config.RetryTimes; i++ {
        if i > 0 {
            time.Sleep(ss.config.RetryInterval)
            global.GVA_LOG.Warn("摘要生成重试", zap.Int("attempt", i+1), zap.Error(lastErr))
        }

        summary, err := ss.aiService.generateText(ctx, []*schema.Message{
            schema.UserMessage(prompt),
        })
        if err == nil {
            return strings.TrimSpace(summary), nil
        }
        lastErr = err
    }
    return "", fmt.Errorf("重试%d次后仍失败: %w", ss.config.RetryTimes, lastErr)
}

// validateSummary 验证摘要质量
func (ss *SummaryService) validateSummary(summary string) error {
    if strings.TrimSpace(summary) == "" {
        return errors.New("摘要为空")
    }

    // Token数量验证
    tokenCount, _ := ss.tokenizer.Count(summary)
    if tokenCount < ss.config.MinSummaryTokens {
        return fmt.Errorf("摘要过短: %d tokens", tokenCount)
    }
    if tokenCount > ss.config.MaxSummaryTokens {
        return fmt.Errorf("摘要过长: %d tokens", tokenCount)
    }

    // 语义完整性检查（检查是否包含常见截断迹象）
    if strings.HasSuffix(strings.TrimSpace(summary), "...") ||
       strings.HasSuffix(strings.TrimSpace(summary), "未完待续") ||
       strings.HasSuffix(strings.TrimSpace(summary), "详细内容") {
        return errors.New("摘要可能被截断")
    }

    return nil
}

// fallbackSummary 降级摘要策略
func (ss *SummaryService) fallbackSummary(summary string) string {
    truncated := make([]rune, 0, len(summary))
    tokenCount := 0

    for _, r := range summary {
        if tokenCount >= ss.config.MaxSummaryTokens {
            break
        }
        truncated = append(truncated, r)
        token++
    }

    string(truncated)
}

// GenerateDeepSummary 生成深度摘要
func (ss *SummaryService) GenerateDeepSummary(ctx context.Context, sessionID int64) (string, error) {
    var session model.Session
    if err := global.GVA_DB.Where("id = ?", sessionID).First(&session).Error; err != nil {
        return "", err
    }

    if session.Summary == "" {
        return "", errors.New("基础摘要不存在")
    }

    prompt := fmt.Sprintf(`请基于以下对话摘要，生成一份更精简的深度摘要。

字数控制在200字以内：

%s`, session.Summary)

    deepSummary, err := ss.aiService.generateText(ctx, []*schema.Message{
        schema.UserMessage(prompt),
    })

    return strings.TrimSpace(deepSummary), err
}
```

```go
// Chat 处理与 AI 模型的单次对话（重构版本）
func (s *AIService) Chat(ctx context.Context, userId uint, req dto.AIChatReq) (<-chan dto.ChatStreamChunk, int64, int64, error) {
    // 1. 初始化智能体
    chatAgent, err := s.newChatAgent(ctx, userId)
    if err != nil {
        return nil, 0, 0, err
    }

    // 2. 会话管理
    var session model.Session
    db := global.GVA_DB.WithContext(ctx)
    isNewSession := false

    if req.SessionID == 0 {
        isNewSession = true
        session = model.Session{
            UserID:  int64(userId),
            Title:   "新对话",
            ModelID: global.GVA_CONFIG.AI.Model,
        }
        if err := db.Create(&session).Error; err != nil {
            global.GVA_LOG.Error("Failed to create AI session", zap.Error(err))
            return nil, 0, 0, errors.New("创建会话失败")
        }
    } else {
        if err := db.Where("id = ? AND user_id = ?", req.SessionID, userId).First(&session).Error; err != nil {
            if errors.Is(err, gorm.ErrRecordNotFound) {
                return nil, 0, 0, errors.New("会话不存在或无权访问")
            }
            return nil, 0, 0, errors.New("查询会话失败")
        }
    }

    // 3. 构建上下文（使用新的上下文管理器）
    userPrompt := s.buildUserPrompt(req)

    // 获取会话级策略或使用全局策略
    strategy := session.GetContextStrategy()
    if strategy == nil {
        strategy = &s.config.ContextStrategy
    }

    // 使用上下文管理器构建上下文
    contextResult, err := s.contextManager.BuildChatContext(ctx, session.ID, userPrompt, 0, strategy)
    if err != nil {
        global.GVA_LOG.Error("Build chat context failed", zap.Error(err))
        return nil, 0, 0, errors.New("构建上下文失败")
    }

    // 4. 保存用户消息
    userMsg := model.Message{
        SessionID: session.ID,
        ParentID:  req.ParentID,
        Role:      string(schema.User),
        Content:   req.Prompt,
        TokensUsed:    0,
        TokenEstimated: int(contextResult.TotalTokens / 2), // 粗略估算
    }
    db.Create(&userMsg)

    // 5. 创建AI回复消息占位符
    aiMsg := model.Message{
        SessionID: session.ID,
        ParentID:  &userMsg.ID,
        Role:      string(schema.Assistant),
        Content:   "",
        TokensUsed:    0,
        TokenEstimated: 0,
    }
    db.Create(&aiMsg)

    // 6. 设置工具事件和当前页面
    msgChan := make(chan dto.ChatStreamChunk, 64)
    agentCtx := withAIToolEventSender(ctx, func(content string) {
        msgChan <- dto.ChatStreamChunk{Type: "tool", Content: content}
    })
    agentCtx = withAICurrentPageURL(agentCtx, req.CurrentPageURL)

    // 7. 启动流式响应
    go func() {
        defer close(msgChan)
        streamResp, err := chatAgent.Stream(agentCtx, contextResult.Messages)

        streamMsgs := make([]*schema.Message, 0, 32)
        defer streamResp.Close()

        for {
            chunk, err := streamResp.Recv()
            if err == io.EOF {
                break
            }
            if err != nil {
                s.handleStreamError(msgChan, err, userMsg.ID, aiMsg.ID, session, isNewSession)
                return
            }

            streamMsgs = append(streamMsgs, chunk)
            s.emitMessageChunk(msgChan, chunk)
        }

        if len(streamMsgs) == 0 {
            s.handleStreamError(msgChan, errors.New("AI未生成最终答案"), userMsg.ID, aiMsg.ID, session, isNewSession)
            return
        }

        // 8. 合并并保存回复
        finalResp, err := schema.ConcatMessages(streamMsgs)
        if err != nil {
            s.handleStreamError(msgChan, errors.New("AI流式结果合并失败"), userMsg.ID, aiMsg.ID, session, isNewSession)
            return
        }

        messageText := s.extractMessageText(finalResp)
        if strings.TrimSpace(messageText) == "" {
            s.handleStreamError(msgChan, errors.New("AI未生成最终答案"), userMsg.ID, aiMsg.ID, session, isNewSession)
            return
        }

        // 更新AI消息内容和Token统计
        aiMsgToken, _ := s.contextManager.GetTokenizer().Count(messageText)
        global.GVA_DB.Model(&model.Message{}).Where("id = ?", aiMsg.ID).Updates(map[string]interface{}{
            "content":         messageText,
            "tokens_used":     aiMsgToken,
            "token_estimated": aiMsgToken,
        })

        // 9. 异步处理摘要更新
        s.finalizeChatSideEffectsAsync(db, session, isNewSession, req.Prompt, userMsg.ID)
    }()

    return msgChan, session.ID, aiMsg.ID, nil
}

// handleStreamError 统一处理流式错误
func (s *AIService) handleStreamError(msgChan chan<- dto.ChatStreamChunk, err error, userMsgID, aiMsgID int64, session model.Session, isNewSession bool) {
    global.GVA_LOG.Error("AI stream error", zap.Error(err))
    msgChan <- dto.ChatStreamChunk{Type: "tool", Content: fmt.Sprintf("AI执行失败: %v", err)}

    // 清理失败的会话和消息
    global.GVA_DB.Unscoped().Where("id IN ?", []int64{userMsgID, aiMsgID}).Delete(&model.Message{})
    if isNewSession {
        global.GVA_DB.Unscoped().Where("id = ?", session.ID).Delete(&model.Session{})
    }
}

// finalizeChatSideEffectsAsync 异步处理副作用（摘要更新等）
func (s *AIService) finalizeChatSideEffectsAsync(db *gorm.DB, session model.Session, isNewSession bool, prompt string, latestUserMsgID int64) {
    go func(sId int64, p string, uId int64, sess model.Session, isNS bool) {
        // 新会话：生成标题
        if isNS {
            s.generateSessionTitle(sId, p)
            return
        }

        // 老会话：判断是否需要更新摘要
        if s.summaryService.ShouldUpdateSummary(context.Background(), db, sess, uId) {
            task := SummaryGenerationTask{
                SessionID:        sess.ID,
                LastMessageID:    sess.LastSummaryMessageID,
                PreviousSummary:  sess.Summary,
                PreviousVersion:  sess.SummaryVersion,
                TaskID:           uuid.New().String(),
            }

            summary, tokenCount, err := s.summaryService.GenerateSummary(context.Background(), task)
            if err == nil {
                global.GVA_DB.Model(&model.Session{}).Where("id = ?", sess.ID).Updates(map[string]interface{}{
                    "summary":                 summary,
                    "summary_token_count":     tokenCount,
                    "last_summary_message_id": uId,
                    "last_summary_at":         time.Now(),
                    "summary_version":         sess.SummaryVersion + 1,
                })

                // 判断是否需要深度摘要
                if sess.SummaryVersion >= s.config.DeepSummaryAfter {
                    deepSum, err := s.summaryService.GenerateDeepSummary(context.Background(), sess.ID)
                    if err == nil && deepSum != "" {
                        global.GVA_DB.Model(&model.Session{}).Where("id = ?", sess.ID).Update("deep_summary", deepSum)
                    }
                }
            }
        }
    }(session.ID, prompt, latestUserMsgID, session, isNewSession)
}

// generateSessionTitle 生成会话标题
func (s *AIService) generateSessionTitle(sessionID int64, prompt string) {
    ctx, cancel := context.WithTimeout(context.Background(), 30*time.Second)
    defer cancel()

    titlePrompt := fmt.Sprintf("请根据以下用户的提问，生成一个简短的对话标题（不超过15个字），不要包含任何标点符号：\n\n%s", prompt)
    title, err := s.generateText(ctx, []*schema.Message{
        schema.UserMessage(titlePrompt),
    })
    if err == nil && title != "" {
        global.GVA_DB.Model(&model.Session{}).Where("id = ?", sessionID).Update("title", title)
    }
}
```

```go
package monitor

import (

// ContextMetrics 上下文指标
type ContextMetrics struct {
    // Token使用
    TotalTokensUsed       int64
    AvgTokensPerMessage   float64
    MaxTokensPerMessage   int

    // 摘要相关
    SummaryCount          int
    SummaryGenDuration    []time.Duration
    SummaryRetries        int
    DeepSummaryCount      int

    // 上下文命中
    ContextHitRate        float64
    AvgContextRounds      float64
    MaxContextRounds      int

    // 错误统计
    ContextBuildErrors    int
    SummaryGenErrors      int
}

// GlobalContextMetrics 全局指标
var GlobalContextMetrics = &ContextMetrics{}

// RecordTokenUsage 记录Token使用
func (m *ContextMetrics) RecordTokenUsage(tokens int) {
    m.TotalTokensUsed += int64(tokens)
    if tokens > m.MaxTokensPerMessage {
        m.MaxTokensPerMessage = tokens
    }
}

// RecordSummaryGeneration 记录摘要生成
func (m *ContextMetrics) RecordSummaryGeneration(duration time.Duration, retryCount int) {
    m.SummaryCount++
    m.SummaryGenDuration = append(m.SummaryGenDuration, duration)
    m.SummaryRetries += retryCount
}

// RecordContextError 记录上下文错误
func (m *ContextMetrics) RecordContextError(errorType string) {
    switch errorType {
    case "build":
        m.ContextBuildErrors++
    case "summary":
        m.SummaryGenErrors++
    }
}

// GetMetricsString 获取指标字符串（用于日志）
func (m *ContextMetrics) GetMetricsString() string {
    return fmt.Sprintf(
        "Tokens[Total:%d, AvgMsg:%.1f, MaxMsg:%d] "+
        "Summary[Count:%d, AvgDuration:%v, Retries:%d] "+
        "Context[HitRate:%.1f%%, AvgRounds:%.1f, MaxRounds:%d] "+
        "Errors[Build:%d, Summary:%d]",
        m.TotalTokensUsed,
        m.AvgTokensPerMessage,
        m.MaxTokensPerMessage,
        m.SummaryCount,
        m.AvgSummaryDuration(),
        m.SummaryRetries,
        m.ContextHitRate*100,
        m.AvgContextRounds,
        m.MaxContextRounds,
        m.ContextBuildErrors,
        m.SummaryGenErrors,
    )
}

func (m *ContextMetrics) AvgSummaryDuration() time.Duration {
    if len(m.SummaryGenDuration) == 0 {
        return 0
    }
    var total time.Duration
    for _, d := range m.SummaryGenDuration {
        total += d
    }
    return total / time.Duration(len(m.SummaryGenDuration))
}
```
